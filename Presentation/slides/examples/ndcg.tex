\begin{align*}
\text{DCG}_k = \sum_{n=1}^{k}\frac{\textit{rel}_n}{\log_2(n + 1)}
\end{align*}
\begin{align*}
\text{nDCG}_k = \frac{\text{DCG}_k}{\text{IDCG}_k}
\end{align*}

\end{column}
